1function cones_location = processLIDAR(LIDAR,rowrow)
  2
  3        
  4    % CONVERT READ DATA
  5    % MATRIX to POINTCLOUD
  6    %         LIDAR = pointCloud(LIDAR);
  7
  8    found = 0;
  9    mark = zeros(30,3);  % COORDINATES OF CONES
 10    ids = zeros(30);   % indices of point that is part of cone
 11
 12    % traverse through all LIDAR DATA
 13    % LIDAR.Location(:,:) = matrix N x 3
 14    % N = LIDAR.Cound <=> number of points (rows)        
 15    % points = current row
 16%     [rowrow columncolumn] = size(LIDAR);
 17    for points=1:rowrow
 18            % current point that we examine
 19            xx = LIDAR(points,1);
 20            yy = LIDAR(points,2);
 21            zz = LIDAR(points,3);
 22
 23            % there are 2 different approaches
 24            % points closer than middle_boundry [in meters]
 25            % points further than middle_boundry [in metters]
 26            % y_boundry makes tunnel vision ylim[-y_boundry +y_boundry]
 27            outter_boundry = 10;
 28            middle_boundry = 7;
 29            inner_boundry = sqrt(2);
 30            back_boundry = -2;
 31            y__inner_boundry = 4;
 32            y__outter_boundry = 4;
 33
 34            % next line is commented in order to check results if this
 35            % constraint is being added to the following if statement            
 36            % (180/pi)*atan(abs(yy)/xx)<65 && 
 37
 38            % CONSTRAINTS:
 39            % 1 - points only infront (X>0)
 40            % 2 - y_boundry tunnel vision
 41            % 3 - omit points closer than 1.42m (that is vehicle itself)
 42            % 4 - saturate furtherest points to 10m
 43            if(xx>back_boundry && sqrt(xx^2+yy^2)>inner_boundry && xx<outter_boundry)
 44
 45                % IF
 46                %   point is closer than middle_boundry
 47                % ELSE
 48                %   point is further than middle_boundry
 49                % END
 50                if (zz>-0.1 && sqrt(xx^2+yy^2)<middle_boundry && abs(yy)<y__inner_boundry)
 51                    % in inner circle consider only points higher
 52                    % than -0.1m
 53                    % LIDAR is approx. posted at the height of 20cm
 54
 55                    Lidar_z = LIDAR(:,3);
 56                    z_02 = find(Lidar_z<0.2);
 57                    locs = z_02(find(Lidar_z(z_02)>-.1));          
 58                    % locs are points which are -10cm < heigth < 20cm
 59                    % height of the cone is not bigger than 30cm
 60
 61%                     x = LIDAR.Location(locs,1);
 62%                     y = LIDAR.Location(locs,2);
 63%                     z = LIDAR.Location(locs,3);
 64%                     LIDAR_close = pointCloud([x,y,z]);
 65
 66                    % find all points which are closer than 50cm from the
 67                    % examined point
 68                    [indices,dists] = findInRadius(LIDAR(locs,:),[xx yy zz],.5,points,1);
 69
 70                    % check if the point is not alone and if points are
 71                    % concentrated over small space
 72                    if(indices(1)~=0 && mean(dists)<.1)                        
 73                        % CONE HAS BEEN FOUND in INNER CIRCLE
 74
 75                        % check if we already marked this cone
 76                        [member,ind] = ismember(ids,locs(indices));
 77                        if(sum(ind)==0)
 78                            found = found+1;
 79                            ids(found) = points;                            
 80                            mark(found,:) = mean(LIDAR(locs(indices),:)); % mark = [mark; xx yy zz];
 81                            % CONE HAS BEEN MARKED
 82                        end
 83                    end 
 84                else
 85                    if(zz<0.2 && sqrt(xx^2+yy^2)>middle_boundry && sqrt(xx^2+yy^2)<outter_boundry && abs(yy)<y__outter_boundry)
 86                        % in outter circle consider only points lower
 87                        % than 0.2m in case that there is a downhill on the
 88                        % track
 89                        % LIDAR is approx. posted at the height of 20cm
 90
 91                        % find all points which are closer than 1m from the
 92                        % examined point
 93                        data_set = [LIDAR(:,1),LIDAR(:,2),LIDAR(:,3)];
 94                        [indices,dists] = findInRadius(data_set,[xx yy zz],1,points,0);
 95                        % check if there is no more than 15 points and if
 96                        % the point is not alone
 97                        % mean variance of their distance must be smaller
 98                        % than 10cm
 99                        if(indices(1)~=0 && mean(dists)<.1)                        
100                            % CONE HAS BEEN FOUND in INNER CIRCLE
101
102                            % check if we already marked this cone
103                            [member,ind] = ismember(ids,indices);
104                            if(sum(ind)==0)
105                                found = found+1;
106                                ids(found) = points;
107                                mark(found,:) = mean(LIDAR(indices,:)); % mark = [mark; xx yy zz];
108                                % CONE HAS BEEN MARKED
109                            end
110                        end 
111                    end
112                end
113            end
114            if(found==30)
115                break;
116            end
117    end
118    if(found>1)
119        cones_location = mark(1:found,:);
120    else
121        cones_location = [];
122    end
123end
